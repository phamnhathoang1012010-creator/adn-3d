const container = document.getElementById("scene"); // Scene const scene = new THREE.Scene(); // Camera const camera = new THREE.PerspectiveCamera( 75, window.innerWidth / window.innerHeight, 0.1, 1000 ); camera.position.z = 30; // Renderer (ĐÂY LÀ CHỖ HAY QUÊN) const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); renderer.setPixelRatio(window.devicePixelRatio); container.appendChild(renderer.domElement); // Ánh sáng const light = new THREE.PointLight(0xffffff, 1); light.position.set(10, 10, 10); scene.add(light); // ADN demo (xoắn đơn giản) const group = new THREE.Group(); for (let i = 0; i < 100; i++) { const geometry = new THREE.SphereGeometry(0.3, 16, 16); const material = new THREE.MeshStandardMaterial({ color: 0x00ffff }); const sphere = new THREE.Mesh(geometry, material); const angle = i * 0.3; sphere.position.set( Math.cos(angle) * 5, i * 0.15 - 7, Math.sin(angle) * 5 ); group.add(sphere); } scene.add(group); // Xoay bằng tay (mobile) let isTouch = false; let lastX = 0; container.addEventListener("touchstart", e => { isTouch = true; lastX = e.touches[0].clientX; }); container.addEventListener("touchmove", e => { if (!isTouch) return; const x = e.touches[0].clientX; const delta = (x - lastX) * 0.01; group.rotation.y += delta; lastX = x; }); container.addEventListener("touchend", () => { isTouch = false; }); // Resize window.addEventListener("resize", () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); }); // Animate function animate() { requestAnimationFrame(animate); renderer.render(scene, camera); } animate();
